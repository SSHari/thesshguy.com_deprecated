# Multi-line Ellipses Demo

To set <TopicText>ellipses</TopicText> at a given line for long content, the following CSS can be used.

```css
.multi-line-ellipses {
  display: -webkit-box;
  -webkit-box-orient: vertical;
  -webkit-line-clamp: 4;
  overflow: hidden;
}
```

export const MultiLineEllipsesDemo = () => {
  const [text, setText] = React.useState(
    'Lorem ipsum dolor sit amet, consectetur adipiscing elit. Vestibulum et nulla sit amet libero lacinia lacinia. Curabitur quis malesuada dui.',
  );
  const [lineClamp, setLineClamp] = React.useState(4);
  return (
    <div className="max-w-full">
      <p className="font-medium text-center mb-6">
        Update the text and line clamp to see how it affects the content.
      </p>
      <div className="flex">
        <div>
          <label className="flex flex-col">
            <span className="font-medium">Line Clamp</span>
            <input
              min="2"
              max="10"
              className="border border-solid border-gray-900 rounded px-4"
              value={lineClamp}
              onChange={(event) => setLineClamp(event.target.value)}
              type="number"
            />
          </label>
          <label className="flex flex-col">
            <span className="font-medium">Text Content</span>
            <input
              className="border border-solid border-gray-900 rounded px-4"
              value={text}
              onChange={(event) => setText(event.target.value)}
            />
          </label>
        </div>
        <div>
          <div
            style={{
              '--lineClamp': lineClamp,
              display: '-webkit-box',
              '-webkitBoxOrient': 'vertical',
              '-webkitLineClamp': 'var(--lineClamp)',
              overflow: 'hidden',
            }}
            className="ml-6"
          >
            {text}
          </div>
        </div>
      </div>
    </div>
  );
};

<MultiLineEllipsesDemo />
